SELECT *,
CONCAT(IMEI,'SI') AS CRUCE_CONTROLNET,
(SELECT LOCAL FROM personal_ppto_vs_real where NAE = USUARIO_FINAL) AS LOCALES,
(SELECT EJECUTIVO FROM personal_ppto_vs_real where NAE = USUARIO_FINAL) AS EJECUTIVO,
CONCAT(MOVIMIENTO, (CASE
                    	WHEN TIPO_VENTA = 'CONTADO' THEN ' CONTADO'
                    	ELSE ' FINANCIADO'
       				END)
      ) AS 'TIPO ALTA',
DAY(FECHA_FACTURA) AS DIA,
TIPO_DOCUMENTO,
FORMAS_DE_PAGO,
CONCAT( (CASE
	WHEN TIPO_DOCUMENTO = 'Factura Contado' THEN 'FC'
	WHEN TIPO_DOCUMENTO = 'Factura Afecta Cuota' THEN 'FAC'
	ELSE 'NT'
END) ,' ',
       (CASE
			WHEN TIPO_VENTA = 'CONTADO' THEN ' CONTADO'
            ELSE ' FINANCIADO'
       	END),
       (CASE
        	WHEN FORMAS_DE_PAGO = 'TARJETAS' THEN 'TAR'
        	WHEN FORMAS_DE_PAGO = 'CORRIENTE' THEN 'COR'
        	ELSE CONCAT(' DIRECTO ',SUBSTRING(FORMAS_DE_PAGO,1,2))
        END)
      ) AS 'TIPO ALTA COMISIONES'
FROM terminales;
